1. Get the Grafana URL by running these commands:
{{- if contains "NodePort" .Values.kubePrometheusStackGrafana.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ template "kube-ceems.namespace" . }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ template "kube-ceems.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ template "kube-ceems.namespace" . }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.kubePrometheusStackGrafana.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get svc -w {{ template "kube-ceems.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ template "kube-ceems.namespace" . }} {{ template "kube-ceems.fullname" . }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
  echo http://$SERVICE_IP:{{ .Values.kubePrometheusStackGrafana.service.port }}
{{- else if contains "ClusterIP" .Values.kubePrometheusStackGrafana.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ template "kube-ceems.namespace" . }} -l "app.kubernetes.io/name=grafana,app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  echo "Visit http://127.0.0.1:80 to access Grafana"
  kubectl port-forward --namespace {{ template "kube-ceems.namespace" . }} $POD_NAME 80
{{- end }}
